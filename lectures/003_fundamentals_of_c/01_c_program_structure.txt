#include <stdio.h>

int main() 
{
    printf("Hello World");
    return 0;
}

* ฟังก์ชัน ใช้สำหรับกลุ่มของชุดคำสั่งไว้ข้างใน
* main function จะเป็นฟังก์ชันแรกของโปรแกรมที่จะถูกเรียกใช้งาน เมื่อโปรแกรมถูกสั่งให้ทำงาน
* { ใช้กำหนดจุดเริ่มต้นของชุดคำสั่ง || } ใช้กำหนดจุดสิ้นสุดของชุดคำสั่ง 
* สาเหตุที่ใน main ต้อง return 0 เพื่อเป็นการบอก compiler ว่าโปรแกรมสามารถทำงานได้ถูกต้อง
